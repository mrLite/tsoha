Redirects to project unless the logged in user is either the creator or an administrator of the project, in which case returns true 

     # File app/controllers/projects_controller.rb, line 105
105:   def can_modify?
106:     project = Project.find(params[:id])
107:     
108:     if logged_in?
109:       user = User.find(session[:user_id])
110:       if user.projects.include?(project)
111:         member = project.members.find_by_user_id(user)
112:         if (member.member_role.role == "creator" or member.member_role.role == "administrator")
113:           return true
